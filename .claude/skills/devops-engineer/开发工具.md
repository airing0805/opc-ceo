# 开发工具配置

## VS Code 配置

### 安装 VS Code

```powershell
# 使用 winget
winget install Microsoft.VisualStudioCode

# 使用 Chocolatey
choco install vscode -y

# 使用 Scoop
scoop bucket add extras
scoop install vscode
```

### 命令行使用

```bash
# 打开文件
code file.txt

# 打开目录
code .

# 打开新窗口
code -n .

# 对比文件
code -d file1.txt file2.txt

# 安装扩展
code --install-extension ms-python.python

# 列出扩展
code --list-extensions

# 卸载扩展
code --uninstall-extension ms-python.python
```

## Git 配置

### 基础配置

```bash
# 用户信息
git config --global user.name "Your Name"
git config --global user.email "your@email.com"

# 默认分支名
git config --global init.defaultBranch main

# 换行符处理
git config --global core.autocrlf input    # Windows
git config --global core.autocrlf false    # Linux/Mac

# 编辑器
git config --global core.editor "code --wait"
```

### 常用别名

```bash
# 基础别名
git config --global alias.st status
git config --global alias.co checkout
git config --global alias.br branch
git config --global alias.ci commit

# 高级别名
git config --global alias.lg "log --oneline --graph --all -20"
git config --global alias.last "log -1 HEAD"
git config --global alias.unstage "reset HEAD --"
git config --global alias.undo "checkout --"
git config --global alias.stash-all "stash save --include-untracked"
git config --global alias.cm "commit -m"
git config --global alias.amend "commit --amend --no-edit"

# 查看别名
git config --global --get-regexp alias
```

### 凭据管理

```bash
# Windows 凭据管理器
git config --global credential.helper manager

# 缓存凭据（15分钟）
git config --global credential.helper cache

# 缓存凭据（自定义时间）
git config --global credential.helper "cache --timeout=3600"

# 存储凭据（明文，不推荐）
git config --global credential.helper store
```

### SSH 密钥配置

```bash
# 生成密钥
ssh-keygen -t ed25519 -C "your@email.com"

# 或 RSA
ssh-keygen -t rsa -b 4096 -C "your@email.com"

# 添加到 agent
eval "$(ssh-agent -s)"
ssh-add ~/.ssh/id_ed25519

# 查看公钥（添加到 GitHub）
cat ~/.ssh/id_ed25519.pub

# 测试连接
ssh -T git@github.com
```

## 环境变量配置脚本

```powershell
# set-env.ps1 - 设置开发环境变量

# 添加到 PATH
function Add-ToPath {
    param([string]$Path)
    $currentPath = [Environment]::GetEnvironmentVariable("PATH", "User")
    if ($currentPath -notlike "*$Path*") {
        [Environment]::SetEnvironmentVariable("PATH", "$currentPath;$Path", "User")
        Write-Host "Added $Path to PATH"
    }
}

# 常用路径
Add-ToPath "C:\Tools\bin"
Add-ToPath "$env:USERPROFILE\.local\bin"
Add-ToPath "$env:USERPROFILE\go\bin"
Add-ToPath "$env:APPDATA\npm"

# 设置开发环境变量
[Environment]::SetEnvironmentVariable("GOPATH", "$env:USERPROFILE\go", "User")
[Environment]::SetEnvironmentVariable("PYTHONIOENCODING", "utf-8", "User")
[Environment]::SetEnvironmentVariable("NODE_OPTIONS", "--max-old-space-size=4096", "User")

Write-Host "Environment variables set. Please restart terminal."
```
